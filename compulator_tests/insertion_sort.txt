int i = 1,j,key;
real b[10] = {0,4,1,8, 3,2,5, 1,11,26};
while(i < 10) {
	key = b[i];
	j = i - 1;
	while(j >= 0 && b[j] > key) {
		b[j+1] = b[j];
		j = j - 1;
	}
	b[j+1] = key; 
}

i = 0;
while ( i < 10) { print(b[i]); }



// Ожидаемые результаты
// на вход 0,4,1,8,3,2,5, 1,11,26
// на выход 0,1,1,2,3,4,5,8,11,26
